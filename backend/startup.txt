# Azure App Service Startup Command
# This file is used by Azure App Service to determine how to start the Python application
# For Linux App Service, this is executed directly
# For Windows App Service, see web.config

# Production: Use Gunicorn with Uvicorn workers
gunicorn app:app --worker-class uvicorn.workers.UvicornWorker --workers 2 --bind 0.0.0.0:8000 --timeout 120 --access-logfile - --error-logfile -
